        <script>
        function rozwiazUlamki() {
            // Wczytanie wartości zmiennej m
            let m_kow = prompt("Podaj wartość m (liczba naturalna większa od 0):");

            // Konwersja na liczbę
            m_kow = parseInt(m_kow);

            if (isNaN(m_kow) || m_kow <= 0) {
                document.write("Podano nieprawidłową wartość. Spróbuj ponownie.");
                return;
            }

            // Obliczanie mianowników według wzoru
            let mianownik1 = m_kow + 1; // Pierwszy mianownik
            let mianownik2 = m_kow * mianownik1; // Drugi mianownik

            // Wyświetlenie wyniku
            document.write("<h2>Rozkład ułamka 1/" + m_kow + ":</h2>");
            document.write("1/" + m_kow + " = 1/" + mianownik1 + " + 1/" + mianownik2);
        }
        </script>
